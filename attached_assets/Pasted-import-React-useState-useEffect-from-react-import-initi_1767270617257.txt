import React, { useState, useEffect } from 'react';
import { initializeApp } from 'firebase/app';
import { 
  getFirestore, doc, onSnapshot, setDoc, query, collection, orderBy, limit 
} from 'firebase/firestore';
import { getAuth, onAuthStateChanged, signInAnonymously } from 'firebase/auth';
import { Target, Flame, Shield, CheckCircle2, Archive, Lock, Brain } from 'lucide-react';

// --- INFRAESTRUCTURA BASADA EN VARIABLES DE ENTORNO (SECRETS) ---
const firebaseConfig = {
  apiKey: import.meta.env.VITE_FIREBASE_API_KEY,
  authDomain: import.meta.env.VITE_FIREBASE_AUTH_DOMAIN,
  projectId: import.meta.env.VITE_FIREBASE_PROJECT_ID,
  storageBucket: import.meta.env.VITE_FIREBASE_STORAGE_BUCKET,
  messagingSenderId: import.meta.env.VITE_FIREBASE_MESSAGING_SENDER_ID,
  appId: import.meta.env.VITE_FIREBASE_APP_ID
};

const app = initializeApp(firebaseConfig);
const auth = getAuth(app);
const db = getFirestore(app);
const appId = 'sistemicar-v2-5';

export default function App() {
  const [user, setUser] = useState(null);
  const [activeTab, setActiveTab] = useState('consola');
  const [userData, setUserData] = useState({ rango: 'Iniciado', esperanza: 50, soberania: 10 });
  const [pasoJefe, setPasoJefe] = useState(null);

  // 1. AUTENTICACIÓN PROFESIONAL
  useEffect(() => {
    const unsubscribe = onAuthStateChanged(auth, async (u) => {
      if (!u) {
        try { await signInAnonymously(auth); } catch (e) { console.error("Error Auth"); }
      }
      setUser(u);
    });
    return () => unsubscribe();
  }, []);

  // 2. SINCRONIZACIÓN DE NÚCLEO
  useEffect(() => {
    if (!user) return;
    const userRef = doc(db, 'artifacts', appId, 'users', user.uid);
    const unsub = onSnapshot(userRef, (d) => {
      if (d.exists()) setUserData(d.data());
      else setDoc(userRef, { rango: 'Iniciado', esperanza: 50, soberania: 10, createdAt: new Date() });
    });
    
    const qJ = query(collection(db, 'artifacts', appId, 'users', user.uid, 'pasoJefe'), orderBy('createdAt', 'desc'), limit(1));
    const unsubJ = onSnapshot(qJ, (s) => {
      if (!s.empty) setPasoJefe({ id: s.docs[0].id, ...s.docs[0].data() });
    });

    return () => { unsub(); unsubJ(); };
  }, [user]);

  // UI Y COMPONENTES (IGUAL QUE EL ANTERIOR PARA MANTENER LÓGICA)
  // ... (Aquí sigue el resto del código de la interfaz que ya aprobaste)
  
  return (
    <div className="min-h-screen bg-[#020202] text-zinc-400 p-6 font-sans">
      <main className="max-w-md mx-auto pb-24 text-center">
         {/* Aquí va el contenido de Consola, Alquimia y Manifiesto */}
         <h1 className="text-white font-serif italic text-2xl mt-10">Sistemicar v2.5</h1>
         <p className="text-xs text-blue-500 mt-2 uppercase tracking-widest">Esperando Configuración de Secrets...</p>
         
         <nav className="fixed bottom-6 left-6 right-6 flex justify-center">
            <div className="glass px-8 py-4 rounded-full border border-white/10 flex gap-10">
              <button onClick={() => setActiveTab('consola')} className={activeTab === 'consola' ? 'text-blue-400' : 'text-zinc-600'}><Target size={22}/></button>
              <button onClick={() => setActiveTab('alquimia')} className={activeTab === 'alquimia' ? 'text-yellow-500' : 'text-zinc-600'}><Flame size={22}/></button>
              <button onClick={() => setActiveTab('manifiesto')} className={activeTab === 'manifiesto' ? 'text-white' : 'text-zinc-600'}><Shield size={22}/></button>
            </div>
         </nav>
      </main>
      <style>{`.glass { background: rgba(255, 255, 255, 0.02); backdrop-filter: blur(20px); }`}</style>
    </div>
  );
}
