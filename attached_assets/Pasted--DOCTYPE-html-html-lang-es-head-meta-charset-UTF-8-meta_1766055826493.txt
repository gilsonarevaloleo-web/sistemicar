    <script>
        let currentCP = 0;

        function showPage(pageId) {
            document.querySelectorAll('.page-content').forEach(p => p.classList.remove('active-page'));
            document.getElementById(pageId).classList.add('active-page');
            window.scrollTo(0, 0);
        }

        function addPoints(pts) {
            const input = document.getElementById('main-input');
            if(!input.value) return;
            currentCP += pts;
            document.getElementById('total-cp').textContent = currentCP;
            
            if(pts > 0) {
                const hope = document.getElementById('hope-list');
                if(hope.innerHTML.includes('No hay')) hope.innerHTML = '';
                const item = document.createElement('div');
                item.className = 'p-3 bg-white/[0.03] border-l-2 border-yellow-500 rounded-lg animate-slide-in';
                item.innerHTML = `<p class="text-[10px] leading-tight italic">"${input.value}"</p>`;
                hope.prepend(item);
            }
            input.value = '';
        }

        function setBoss() {
            const input = document.getElementById('boss-input');
            if(!input.value) return;
            const display = document.getElementById('boss-display');
            display.innerHTML = `
                <h4 class="text-2xl font-black italic text-white">${input.value.toUpperCase()}</h4>
                <div class="flex gap-4">
                    <button onclick="resolveBoss(true)" class="flex-1 py-4 bg-yellow-500 text-black font-black rounded-xl text-[10px] uppercase">Vencido (+100)</button>
                    <button onclick="resolveBoss(false)" class="px-6 border border-white/10 text-slate-500 rounded-xl text-[9px] uppercase">Remplazar</button>
                </div>
            `;
        }

        function resolveBoss(success) {
            if(success) {
                currentCP += 100;
                document.getElementById('total-cp').textContent = currentCP;
            }
            document.getElementById('boss-display').innerHTML = `
                <input id="boss-input" type="text" placeholder="¿Cuál es tu próximo Paso Jefe?" class="w-full bg-transparent text-xl font-bold outline-none border-b border-white/10 pb-2">
                <button onclick="setBoss()" class="w-full py-4 border border-white/10 rounded-2xl text-[10px] font-black uppercase tracking-[0.3em]">Anclar Comando</button>
            `;
        }
    </script>
